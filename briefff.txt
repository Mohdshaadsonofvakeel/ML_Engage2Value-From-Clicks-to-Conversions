def apply_frequency_encoding(train_df, test_df, categorical_cols):
    for col in categorical_cols:
        if col != 'purchaseValue':
            freq_map = train_df[col].value_counts().to_dict()
            train_df[col] = train_df[col].map(freq_map).fillna(0)
